# 组件通信

# 父传子

  • props：用于接收父组件属性值，可以用数组或对象表示。prop 属性传递是单向的，只能在父组件修改属性值，反过来不行！
export default {
  props: {
    status: {
      // type 值可为String,Number,Boolean,Array,Object,Date,Function,Symbol
      // type 还可以是一个自定义的构造函数，并且通过 instanceof 来进行检查确认
      type: String,
      required: true,
      //默认值,对象或数组默认值必须从一个工厂函数获取如 default:()=>[]
      default: '/static/xxx',
      validator:(value) {
        // 这个值必须匹配下列字符串中的一个
        return ['success', 'warning', 'danger'].indexOf(value) !== -1
      }
    }
  },
  data() {
    return {}
  }
}

  • $attrs：收集父组件中除了通过 props 传递的其它所有属性（class 和 style 除外）

    • 可以通过 v-bind:$attrs 向更深组件传值

  • $listeners：将父组件中的 (不含 .native 修饰器的) v-on 监听事件传递给子组件

    • 可以通过 v-bind:$listeners 向更深组件传递

# 子传父

  • $emit：子组件触发父组件绑定的事件
// 父组件
<home @title="title">
// 子组件 title 是事件名
this.$emit('title',[{title:'这是参数}])

# $refs

  • 一个对象，持有已注册过 ref 的所有子组件（官网定义）。
// 父组件, 所有被定义为 dom 或组件的 ref 属性
<home ref="home"/>

mounted(){
  console.log(this.$refs.home) //即可拿到子组件的实例,就可以直接操作 data 和 methods
}

# $root

  • 当前组件树的根 Vue 实例。如果当前实例没有父实例，此实例将会是其自己（官网定义） (opens new window)。

mounted(){
  //获取根实例,最后所有组件都是挂载到根实例上
  console.log(this.$root) 
  //获取根实例的一级子组件
  console.log(this.$root.$children[0]) 
  //获取根实例的二级子组件
  console.log(this.$root.$children[0].$children[0]) 
}

# $parent、$children

  • $parent：父组件
  • $children：当前实例的直接子组件（不保证顺序，也不是响应式的）

# .sync 修饰符

用于使用 prop 后不能双向绑定数据的情况， 因为 prop 只能单向数据传递，有时需要通过子组件修改 prop 属性，可以通过事件绑定的形式触发修改 prop 属性值，如下代码所示：可以用 .sync 改写（事件触发形式的语法糖）

//父组件
<text-document v-bind:title="doc.title" 
   v-on:update:title="doc.title = $event"></text-document>

//子组件
this.$emit('update:title', newTitle)

//.sync 写法
<text-document v-bind:title.sync="doc.title"></text-document>

# provide、inject

  • 依赖注入
//父组件: provide 是一个对象,提供一个属性或方法
provide: { 
  //传递方式为值传递，对象传递地址
  //（父组件修改 title 值，子组件不会改变，如果title 是对象，会响应变动）
  title: 'title',  
  vmMethod:()=>{}
},

// 子或者孙子组件，数组或者对象,注入到子组件
inject: ['title','vmMethod'], //
mounted() {
  this.fooMethod()
  console.log(this.foo)
}

# EventBus

  • 事件总线（发布-订阅设计模式），解决任意组件关系间的通信
// event-bus.js
import Vue from 'vue';
export const EventBus = new Vue();

// A.vue 
<template>
    <button @click="sendMsg()">-</button>
</template>

<script> 
import { EventBus } from "../event-bus.js";
export default {
  methods: {
    sendMsg() {
      EventBus.$emit("aMsg", '来自A页面的消息');
    }
  }
}; 
</script>

//B.vue
<template>
  <p>{{msg}}</p>
</template>

<script> 
import { EventBus } from "../event-bus.js";
export default {
  data(){
    return {
      msg: ''
    }
  },
  mounted() {
    EventBus.$on("aMsg", (msg) => {
      // A发送来的消息
      this.msg = msg;
    });
  }
};
</script>

如果使用不善，EventBus会是一种灾难，开发vue单页应用时，如果你在某一个页面刷新了之后，与之相关的EventBus会被移除，这样就导致业务走不下去。还要就是如果业务有反复操作的页面，EventBus在监听的时候就会触发很多次，也是一个非常大的隐患。这时候我们就需要好好处理EventBus在项目中的关系。通常会用到，在vue页面销毁时，同时移除EventBus事件监听。EventBus.$off() 移除所有事件

# broadcast、dispatch

  • 事件广播、派发
function broadcast(componentName, eventName, params) {
  this.$children.forEach(child => {
    var name = child.$options.componentName;

    if (name === componentName) {
      child.$emit.apply(child, [eventName].concat(params));
    } else {
      broadcast.apply(child, [componentName, eventName].concat(params));
    }
  });
}
export default {
  methods: {
    dispatch(componentName, eventName, params) {
      var parent = this.$parent;
      var name = parent.$options.componentName;
      while (parent && (!name || name !== componentName)) {
        parent = parent.$parent;

        if (parent) {
          name = parent.$options.componentName;
        }
      }
      if (parent) {
        parent.$emit.apply(parent, [eventName].concat(params));
      }
    },
    broadcast(componentName, eventName, params) {
      broadcast.call(this, componentName, eventName, params);
    }
  }
}
